class BoardModelTests {
    describe('Board model', () => {
        it('should create a new board', () => {
            const board: Board = {
            name: 'Test board',
            createdBy: 1,
            updatedBy: 1
            };
            const newBoard = new Board(board);
            expect(newBoard.name).toBe(board.name);
            expect(newBoard.createdBy).toBe(board.createdBy);
            expect(newBoard.updatedBy).toBe(board.updatedBy);
        });
    });
}
