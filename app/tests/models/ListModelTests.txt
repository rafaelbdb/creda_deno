class ListModelTests {
    describe('List model', () => {
        it('should create a new list', () => {
            const list: List = {
            name: 'Test list',
            boardId: 1,
            createdBy: 1,
            updatedBy: 1
            };
            const newList = new List(list);
            expect(newList.name).toBe(list.name);
            expect(newList.boardId).toBe(list.boardId);
            expect(newList.createdBy).toBe(list.createdBy);
            expect(newList.updatedBy).toBe(list.updatedBy);
        });
    });
}
