class CommentModelTests {
    describe('Comment model', () => {
        it('should create a new comment', () => {
            const comment: Comment = {
            text: 'Test comment',
            taskId: 1,
            createdBy: 1,
            updatedBy: 1
            };
            const newComment = new Comment(comment);
            expect(newComment.text).toBe(comment.text);
            expect(newComment.taskId).toBe(comment.taskId);
            expect(newComment.createdBy).toBe(comment.createdBy);
            expect(newComment.updatedBy).toBe(comment.updatedBy);
        });
    });
}
